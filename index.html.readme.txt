        // P√§ivit√§ kello sekunnin v√§lein
        function updateClock() {
            const now = new Date();
            
            // Aika: HH:MM:SS
            const time = now.toLocaleTimeString('fi-FI', {
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });

            // P√§iv√§m√§√§r√§: ke 31.12.2025 (esim.)
            const dateOptions = { weekday: 'short', day: '2-digit', month: '2-digit', year: 'numeric' };
            const date = now.toLocaleDateString('fi-FI', dateOptions)
                           .replace(/,/g, '') // Poistaa pilkut
                           .replace(/\./g, '.'); // Varmistaa pisteet

            document.getElementById('clock').textContent = time;
            document.getElementById('date').textContent = date;
        }

        // Alusta kello heti
        updateClock();

        // P√§ivit√§ joka sekunti
        setInterval(updateClock, 1000);
